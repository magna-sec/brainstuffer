The Formula — extracting byte BN from a 64-bit register

mov rax, rdi
; bring target to B0:
shr rax, N×8       ; N = byte index (0 = LSB)
; wipe everything above B0:
shl rax, 56        ; 56 = 64 − 8  →  push B0 to B7
shr rax, 56        ; slide B7 back to B0
; result: rax = 0x00…00  [target byte]
